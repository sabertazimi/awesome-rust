296fn spawn_watch_shell(
297    failed_exercise_hint: &Arc<Mutex<Option<String>>>,
298    should_quit: Arc<AtomicBool>,
299) {
300    let failed_exercise_hint = Arc::clone(failed_exercise_hint);
301    println!("Welcome to watch mode! You can type 'help' to get an overview of the commands you can use here.");
302    thread::spawn(move || loop {
303        let mut input = String::new();
304        match io::stdin().read_line(&mut input) {
305            Ok(_) => {
306                let input = input.trim();
307                if input == "hint" {
308                    if let Some(hint) = &*failed_exercise_hint.lock().unwrap() {
309                        println!("{hint}");
310                    }
311                } else if input == "clear" {
312                    println!("\x1B[2J\x1B[1;1H");
313                } else if input.eq("quit") {
314                    should_quit.store(true, Ordering::SeqCst);
315                    println!("Bye!");
316                } else if input.eq("help") {
317                    println!("Commands available to you in watch mode:");
318                    println!("  hint  - prints the current exercise's hint");
319                    println!("  clear - clears the screen");
320                    println!("  quit  - quits watch mode");
321                    println!("  help  - displays this help message");
322                    println!();
323                    println!("Watch mode automatically re-evaluates the current exercise");
324                    println!("when you edit a file's contents.")
325                } else {
326                    println!("unknown command: {input}");
327                }
328            }
329            Err(error) => println!("error reading command: {error}"),
330        }
331    });
332}

334fn find_exercise<'a>(name: &str, exercises: &'a [Exercise]) -> &'a Exercise {
335    if name.eq("next") {
336        exercises
337            .iter()
338            .find(|e| !e.looks_done())
339            .unwrap_or_else(|| {
340                println!("🎉 Congratulations! You have done all the exercises!");
341                println!("🔚 There are no more exercises to do next!");
342                std::process::exit(1)
343            })
344    } else {
345        exercises
346            .iter()
347            .find(|e| e.name == name)
348            .unwrap_or_else(|| {
349                println!("No exercise found for '{name}'!");
350                std::process::exit(1)
351            })
352    }
353}

355enum WatchStatus {
356    Finished,
357    Unfinished,
358}
359
360fn watch(exercises: &[Exercise], verbose: bool) -> notify::Result<WatchStatus> {
361    /* Clears the terminal with an ANSI escape code.
362    Works in UNIX and newer Windows terminals. */
363    fn clear_screen() {
364        println!("\x1Bc");
365    }
366
367    let (tx, rx) = channel();
368    let should_quit = Arc::new(AtomicBool::new(false));
369
370    let mut watcher: RecommendedWatcher = Watcher::new(tx, Duration::from_secs(2))?;
371    watcher.watch(Path::new("./exercises"), RecursiveMode::Recursive)?;
372
373    clear_screen();
374
375    let to_owned_hint = |t: &Exercise| t.hint.to_owned();
376    let failed_exercise_hint = match verify(exercises.iter(), verbose) {
377        Ok(_) => return Ok(WatchStatus::Finished),
378        Err(exercise) => Arc::new(Mutex::new(Some(to_owned_hint(exercise)))),
379    };
380    spawn_watch_shell(&failed_exercise_hint, Arc::clone(&should_quit));
381    loop {
382        match rx.recv_timeout(Duration::from_secs(1)) {
383            Ok(event) => match event {
384                DebouncedEvent::Create(b) | DebouncedEvent::Chmod(b) | DebouncedEvent::Write(b) => {
385                    if b.extension() == Some(OsStr::new("rs")) && b.exists() {
386                        let filepath = b.as_path().canonicalize().unwrap();
387                        let pending_exercises = exercises
388                            .iter()
389                            .skip_while(|e| !filepath.ends_with(&e.path))
390                            // .filter(|e| filepath.ends_with(&e.path))
391                            .chain(
392                                exercises
393                                    .iter()
394                                    .filter(|e| !e.looks_done() && !filepath.ends_with(&e.path)),
395                            );
396                        clear_screen();
397                        match verify(pending_exercises, verbose) {
398                            Ok(_) => return Ok(WatchStatus::Finished),
399                            Err(exercise) => {
400                                let mut failed_exercise_hint = failed_exercise_hint.lock().unwrap();
401                                *failed_exercise_hint = Some(to_owned_hint(exercise));
402                            }
403                        }
404                    }
405                }
406                _ => {}
407            },
408            Err(RecvTimeoutError::Timeout) => {
409                // the timeout expired, just check the `should_quit` variable below then loop again
410            }
411            Err(e) => println!("watch error: {e:?}"),
412        }
413        // Check if we need to exit
414        if should_quit.load(Ordering::SeqCst) {
415            return Ok(WatchStatus::Unfinished);
416        }
417    }
418}

420fn rustc_exists() -> bool {
421    Command::new("rustc")
422        .args(["--version"])
423        .stdout(Stdio::null())
424        .spawn()
425        .and_then(|mut child| child.wait())
426        .map(|status| status.success())
427        .unwrap_or(false)
